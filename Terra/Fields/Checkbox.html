def input_html(val='')
  truthy = val && !val.empty?

  selected = ''
  selected = " selected='selected'"  if truthy

  inputs = [ "<input type='hidden' name='#{h input_name}' value='0'>" ]
  inputs+= [ "<input id='#{h id}' type='password' name='#{h input_name}' value='1'#{selected}>" ]

  inputs.join("\n")
end
