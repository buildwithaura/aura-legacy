def path(*a)
  ret = "/#{self.class.class_name}/#{self.id}"
  ret += "/#{a.shift.to_s}"  if a.first.is_a?(String) || a.first.is_a?(Symbol)
  ret += "?" + Aura::Utils.query_string(a.shift)  if a.first.is_a?(Hash)
  ret
end

def sort_index
  pos = nil
  pos ||= self.position  if self.respond_to?(:position)
  [pos || 9999, self.id]
end
