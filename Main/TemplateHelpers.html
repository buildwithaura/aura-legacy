def partial(template, locals={})
  layout, @layout = @layout, nil
  out = show template, {:layout => false}, locals
  @layout = layout
  out
end

def yield_content_of(template, section, locals={})
  custom_activate!

  show template, { :layout => false }, locals
  ret = yield_content(section)

  custom_deactivate!
  ret
end
