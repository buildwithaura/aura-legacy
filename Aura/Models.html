def get(name)
  name = camelize(name)  if name.downcase == name
  begin
    const_get(name)
  rescue NameError
    nil
  end
end

def unpack
  all.each do |model|
    klass = model.name.split('::').last
    Kernel.const_set(klass, model)  unless Kernel.const_defined?(klass)
  end
end
