def flush!(&blk)
  blk = lambda { |*a| }  unless block_given?

  tables = Aura::Models.all.map { |m| m.table_name }
  tables << :schema_info
  tables &= self.db.tables

  tables.each do |table|
    blk.call(:drop_table, table)
    self.db.drop_table table
  end
end
